## שאילתות בדיקה (SQL)

```sql
-- בדוק שכירויות לחשבון נטוביץ
SELECT 
  l.id,
  p.address,
  u.apartmentNumber,
  t.name as tenant,
  l.monthlyRent,
  l.status,
  l.startDate,
  l.endDate
FROM leases l
JOIN units u ON l.unitId = u.id
JOIN properties p ON u.propertyId = p.id
JOIN tenants t ON l.tenantId = t.id
JOIN accounts a ON l.accountId = a.id
WHERE a.name = 'משפחת נטוביץ'
ORDER BY p.address, u.apartmentNumber;

-- ספירת שכירויות לפי סטטוס
SELECT status, COUNT(*) as count
FROM leases l
JOIN accounts a ON l.accountId = a.id
WHERE a.name = 'משפחת נטוביץ'
GROUP BY status;
```
